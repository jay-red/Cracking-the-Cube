avgTrials = []
for a in range( 0, 3 ):
    winners = []
    wins = 0
    loss = 0
    for i in range( 0, 10000 ):
        trialWin = []
        for x in comp:
            temp = np.random.choice( x, size=5 )
            temp = np.delete( temp, [temp.tolist().index( np.amin( temp ) ), temp.tolist().index( np.amax( temp ) ) ] )
            trialWin.append( np.average( temp ) )
        winners.append( trialWin.index( np.amin( trialWin ) ) )
        if avg.index( np.amin( avg ) ) == trialWin.index( np.amin( trialWin ) ):
            wins += 1
        else:
            loss += 1
    avgTrials.append( [wins, loss] )
